# TypeScript Advanced Patterns and Techniques

TypeScript is more than type annotations for JavaScript. Its type system can model real application constraints, make refactors safer, and turn common runtime mistakes into editor feedback. This guide walks through practical advanced patterns you can use in production code.

## Utility Types

### Pick and Omit

Use `Pick` and `Omit` to create focused types from larger domain models.

```typescript
interface User {
  id: string;
  name: string;
  email: string;
  password: string;
  createdAt: Date;
}

// Select only the fields that are safe to expose.
type PublicUser = Pick<User, 'id' | 'name' | 'email'>;

// Remove fields that should be generated by the system.
type CreateUserInput = Omit<User, 'id' | 'createdAt'>;

function getPublicUserInfo(user: User): PublicUser {
  return {
    id: user.id,
    name: user.name,
    email: user.email,
  };
}
```

### Partial and Required

`Partial` is useful for patch updates, while `Required` is useful after defaults have been applied.

```typescript
interface Config {
  apiUrl: string;
  timeout: number;
  retries: number;
}

type PartialConfig = Partial<Config>;
type CompleteConfig = Required<PartialConfig>;

function updateConfig(config: PartialConfig): void {
  // Only update fields that were provided.
}

function withDefaults(config: PartialConfig): CompleteConfig {
  return {
    apiUrl: config.apiUrl ?? 'https://api.example.com',
    timeout: config.timeout ?? 5000,
    retries: config.retries ?? 3,
  };
}
```

## Conditional Types

Conditional types let you choose a type based on another type.

```typescript
type IsString<T> = T extends string ? true : false;

type Test1 = IsString<string>; // true
type Test2 = IsString<number>; // false

type ApiResponse<T> = T extends string
  ? { message: T }
  : T extends number
    ? { code: T }
    : { data: T };
```

They are especially useful when a library API needs to infer different return shapes from different inputs.

```typescript
type EventPayload<TEvent> = TEvent extends 'user.created'
  ? { userId: string; email: string }
  : TEvent extends 'invoice.paid'
    ? { invoiceId: string; amount: number }
    : never;

function publish<TEvent extends 'user.created' | 'invoice.paid'>(
  event: TEvent,
  payload: EventPayload<TEvent>
) {
  return { event, payload };
}
```

## Mapped Types

Mapped types transform every property of an object type.

```typescript
interface User {
  id: string;
  name: string;
  email: string;
}

type ReadonlyUser = {
  readonly [K in keyof User]: User[K];
};

type NullableUser = {
  [K in keyof User]: User[K] | null;
};
```

You can also remap property names with template literal types.

```typescript
type PrefixKeys<T, Prefix extends string> = {
  [K in keyof T as `${Prefix}_${string & K}`]: T[K];
};

type DbUser = PrefixKeys<User, 'user'>;
// { user_id: string; user_name: string; user_email: string }
```

## Template Literal Types

Template literal types are great for route names, event names, and strongly typed string conventions.

```typescript
type Method = 'GET' | 'POST' | 'PUT' | 'DELETE';
type Resource = 'users' | 'posts' | 'comments';

type ApiEndpoint = `${Method} /api/${Resource}`;

const endpoint: ApiEndpoint = 'GET /api/users';
```

You can also extract route parameters:

```typescript
type ExtractRouteParams<T extends string> =
  T extends `${string}:${infer Param}/${infer Rest}`
    ? Param | ExtractRouteParams<Rest>
    : T extends `${string}:${infer Param}`
      ? Param
      : never;

type Params = ExtractRouteParams<'/users/:id/posts/:postId'>;
// 'id' | 'postId'
```

## Function Overloads

Function overloads are useful when one function has multiple valid call shapes.

```typescript
function formatValue(value: string): string;
function formatValue(value: number): string;
function formatValue(value: Date): string;
function formatValue(value: string | number | Date): string {
  if (value instanceof Date) {
    return value.toISOString();
  }

  if (typeof value === 'number') {
    return value.toLocaleString();
  }

  return value.trim();
}
```

## Generic Constraints

Use constraints to require the shape your function actually needs.

```typescript
interface HasId {
  id: string;
}

function indexById<T extends HasId>(items: T[]): Record<string, T> {
  return items.reduce<Record<string, T>>((acc, item) => {
    acc[item.id] = item;
    return acc;
  }, {});
}
```

Constraints keep APIs flexible while preventing invalid inputs.

```typescript
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

const user = { id: 'u_1', email: 'hello@example.com' };
const email = getProperty(user, 'email');
```

## Type Guards

Type guards connect runtime checks with compile-time narrowing.

```typescript
interface Cat {
  kind: 'cat';
  meow: () => void;
}

interface Dog {
  kind: 'dog';
  bark: () => void;
}

type Animal = Cat | Dog;

function isCat(animal: Animal): animal is Cat {
  return animal.kind === 'cat';
}

function handleAnimal(animal: Animal) {
  if (isCat(animal)) {
    animal.meow();
  } else {
    animal.bark();
  }
}
```

Assertion functions are helpful at module boundaries:

```typescript
function assertIsNumber(value: unknown): asserts value is number {
  if (typeof value !== 'number') {
    throw new Error('Expected a number');
  }
}

function double(value: unknown) {
  assertIsNumber(value);
  return value * 2;
}
```

## Practical API Client Example

Here is a compact example that uses several of these ideas together.

```typescript
interface ApiRoutes {
  '/users': {
    GET: { response: User[] };
    POST: { body: CreateUserInput; response: User };
  };
  '/users/:id': {
    GET: { params: { id: string }; response: User };
    DELETE: { params: { id: string }; response: { success: true } };
  };
}

type Route = keyof ApiRoutes;
type RouteMethod<TRoute extends Route> = keyof ApiRoutes[TRoute];
type RouteConfig<
  TRoute extends Route,
  TMethod extends RouteMethod<TRoute>,
> = ApiRoutes[TRoute][TMethod];

async function request<
  TRoute extends Route,
  TMethod extends RouteMethod<TRoute>,
>(
  route: TRoute,
  method: TMethod,
  config: Omit<RouteConfig<TRoute, TMethod>, 'response'>
): Promise<RouteConfig<TRoute, TMethod>['response']> {
  const response = await fetch(route, {
    method: String(method),
    body: 'body' in config ? JSON.stringify(config.body) : undefined,
  });

  return response.json();
}
```

## Summary

Advanced TypeScript is most valuable when it clarifies intent:

1. Use utility types to avoid duplicating object shapes.
2. Use conditional and mapped types when APIs change shape based on input.
3. Use template literal types for string conventions such as routes and events.
4. Use overloads and generic constraints to keep public functions precise.
5. Use type guards at runtime boundaries.

The goal is not to make types clever. The goal is to make invalid states harder to express and ordinary maintenance easier to perform.
